country_event = {
	id = KNT_event.71
	title = KNT_event.71.t
	desc = KNT_event.71.d
	trigger = {
		check_variable = {
			var = KNT_culture_progress
			value = 250
			compare = equals
		} 
        original_tag = KNT			
	}
	mean_time_to_happen = {
	    days = 1
    }
	option = {
		name = KNT_event.71.a
		custom_effect_tooltip = KNT_event71.tt
		hidden_effect = {
			set_variable = {
				KNT_culture_progress = 0
			}
			if = {
				limit = {
					KNT = {
						has_country_flag = KNT_culture_dead_flag
					}
				}
				KNT = {
					set_country_flag = KNT_culture_battered_flag
				}
				KNT = {
					clr_country_flag = KNT_culture_dead_flag
				}
				swap_ideas = {
					remove_idea = KNT_cultural_death
					add_idea = KNT_cultural_battered
				}		
			}
			else_if = {
				limit = {
					KNT = {
						has_country_flag = KNT_culture_battered_flag
					}
				}
				KNT = {
					set_country_flag = KNT_culture_recovering_flag
				}
				KNT = {
					clr_country_flag = KNT_culture_battered_flag
				}
				swap_ideas = {
					remove_idea = KNT_cultural_battered
					add_idea = KNT_cultural_recovering
				}
			}
			else_if = {
				limit = {
					KNT = {
						has_country_flag = KNT_culture_recovering_flag
					}
